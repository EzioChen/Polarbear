## 1. 项目概述

#### 3.1.4 核心算法

**文件收集算法**:
```typescript
async function collectFiles(nodes: FileNode[]): Promise<FileItem[]> {
  const files: FileItem[] = [];
  
  for (const node of nodes) {
    if (node.type === 'file' && node.sourcePath) {
      // 验证文件存在性
      if (await fs.pathExists(node.sourcePath)) {
        files.push({
          sourcePath: node.sourcePath,
          targetPath: node.path,
          size: node.size || 0
        });
      } else {
        throw new FileNotFoundError(node.sourcePath);
      }
    } else if (node.type === 'folder' && node.children) {
      // 递归处理子节点
      const childFiles = await collectFiles(node.children);
      files.push(...childFiles);
    }
  }
  
  return files;
}
```

**ZIP压缩算法**:
```typescript
async function createZipArchive(
  files: FileItem[],
  outputPath: string
): Promise<ZipResult> {
  const archiver = require('archiver')('zip', {
    zlib: { level: 9 }  // 最高压缩级别
  });
  
  const output = fs.createWriteStream(outputPath);
  
  return new Promise((resolve, reject) => {
    output.on('close', () => {
      resolve({
        path: outputPath,
        size: archiver.pointer(),
        fileCount: files.length
      });
    });
    
    archiver.on('error', reject);
    archiver.pipe(output);
    
    // 添加文件到压缩包
    for (const file of files) {
      archiver.file(file.sourcePath, { name: file.targetPath });
    }
    
    archiver.finalize();
  });
}
```

#### 3.2.5 PDF生成方案

**技术选型**: 使用 `puppeteer` + `marked` 实现 Markdown 转 PDF

**实现步骤**:
1. 使用 `marked` 将 Markdown 转换为 HTML
2. 使用 `puppeteer` 将 HTML 渲染为 PDF
3. 应用 CSS 样式模板
4. 输出 PDF 文件

```typescript
async function generatePDF(
  markdown: string,
  outputPath: string,
  template: PDFTemplate
): Promise<PDFResult> {
  const puppeteer = require('puppeteer');
  const marked = require('marked');
  
  // 转换Markdown为HTML
  const htmlContent = marked.parse(markdown);
  
  // 构建完整HTML文档
  const fullHtml = `
    <!DOCTYPE html>
    <html>
    <head>
      <meta charset="UTF-8">
      <style>${template.css}</style>
    </head>
    <body>
      ${htmlContent}
    </body>
    </html>
  `;
  
  // 使用Puppeteer生成PDF
  const browser = await puppeteer.launch();
  const page = await browser.newPage();
  await page.setContent(fullHtml, { waitUntil: 'networkidle0' });
  await page.pdf({
    path: outputPath,
    format: 'A4',
    printBackground: true,
    margin: { top: '2cm', right: '2cm', bottom: '2cm', left: '2cm' }
  });
  await browser.close();
  
  return {
    path: outputPath,
    size: fs.statSync(outputPath).size
  };
}
```

### 4.2 核心控制器 (PublishManager)

```typescript
class PublishManager {
  private state: PublishState = 'IDLE';
  private steps: PublishStep[] = [
    { id: 'check', name: '配置检查', status: 'pending' },
    { id: 'pack', name: '文件打包', status: 'pending' },
    { id: 'generate', name: '文档生成', status: 'pending' },
    { id: 'assemble', name: '邮件组装', status: 'pending' }
  ];
  
  private filePackModule: FilePackModule;
  private docGenModule: DocGenModule;
  private emailAsmModule: EmailAsmModule;
  private cacheStore: CacheStore;
  
  constructor(context: vscode.ExtensionContext) {
    this.cacheStore = new CacheStore(context);
    this.filePackModule = new FilePackModule(this.cacheStore);
    this.docGenModule = new DocGenModule(this.cacheStore);
    this.emailAsmModule = new EmailAsmModule(this.cacheStore);
  }
  
  /**
   * 启动发布流程
   */
  async startPublish(): Promise<void> {
    if (this.state !== 'IDLE') {
      throw new Error('发布流程正在进行中');
    }
    
    this.state = 'CHECKING';
    
    try {
      // 步骤1: 配置检查
      await this.runStep('check', () => this.checkConfigs());
      
      // 步骤2: 文件打包
      this.state = 'PACKAGING';
      const packResult = await this.runStep('pack', () => 
        this.filePackModule.pack()
      );
      
      // 步骤3: 文档生成
      this.state = 'GENERATING';
      const docResult = await this.runStep('generate', () =>
        this.docGenModule.generate()
      );
      
      // 步骤4: 邮件组装
      this.state = 'ASSEMBLING';
      await this.runStep('assemble', () =>
        this.emailAsmModule.assemble({
          zipPath: packResult.path,
          pdfPath: docResult.pdfPath,
          mdPath: docResult.mdPath
        })
      );
      
      this.state = 'COMPLETED';
      
    } catch (error) {
      this.state = 'ERROR';
      throw error;
    }
  }
  
  /**
   * 执行单个步骤
   */
  private async runStep<T>(
    stepId: string,
    executor: () => Promise<T>
  ): Promise<T> {
    const step = this.steps.find(s => s.id === stepId);
    if (!step) throw new Error(`未知步骤: ${stepId}`);
    
    step.status = 'running';
    this.notifyStateChange();
    
    try {
      const result = await executor();
      step.status = 'completed';
      this.notifyStateChange();
      return result;
    } catch (error) {
      step.status = 'failed';
      step.error = error.message;
      this.notifyStateChange();
      throw error;
    }
  }
  
  /**
   * 检查配置文件
   */
  private async checkConfigs(): Promise<ConfigCheckResult> {
    const checks = [
      {
        name: 'publish-config.json',
        path: '.releasePlan/publish-config.json',
        required: true
      },
      {
        name: 'release-notes.json',
        path: '.releasePlan/release-notes.json',
        required: true
      },
      {
        name: 'smtp-config.json',
        path: '.releasePlan/smtp-config.json',
        required: true,
        validate: async () => {
          // 额外验证SMTP连接
          return await this.validateSmtpConnection();
        }
      }
    ];
    
    const results = [];
    for (const check of checks) {
      const exists = await fs.pathExists(check.path);
      let valid = exists;
      let error = null;
      
      if (exists && check.validate) {
        try {
          valid = await check.validate();
        } catch (e) {
          valid = false;
          error = e.message;
        }
      }
      
      results.push({
        ...check,
        exists,
        valid,
        error
      });
    }
    
    const allValid = results.every(r => !r.required || r.valid);
    if (!allValid) {
      throw new ConfigCheckError(results);
    }
    
    return { valid: true, checks: results };
  }
  
  /**
   * 通知状态变更
   */
  private notifyStateChange(): void {
    // 通过webview发送状态更新
    this.webviewPanel?.webview.postMessage({
      type: 'stateUpdate',
      data: {
        state: this.state,
        steps: this.steps,
        progress: this.calculateProgress()
      }
    });
  }
}
```

### 8.1 错误分类

```typescript
// 错误类型枚举
enum ErrorType {

  SMTP_AUTHENTICATION_FAILED = 'SMTP_AUTHENTICATION_FAILED',
  
  // 系统错误
  DISK_SPACE_INSUFFICIENT = 'DISK_SPACE_INSUFFICIENT',
  MEMORY_INSUFFICIENT = 'MEMORY_INSUFFICIENT',
  
  // 流程错误
  STEP_EXECUTION_FAILED = 'STEP_EXECUTION_FAILED',
  STEP_TIMEOUT = 'STEP_TIMEOUT',
  DEPENDENCY_MISSING = 'DEPENDENCY_MISSING'
}

// 错误基类
class PublishError extends Error {
  constructor(
    public type: ErrorType,
    message: string,
    public details?: Record<string, unknown>,
    public recoverable: boolean = false
  ) {
    super(message);
    this.name = 'PublishError';
  }
}
```

### 8.2 错误处理策略

| 错误类型 | 处理策略 | 用户提示 | 恢复选项 |
|----------|----------|----------|----------|
| 配置不存在 | 引导创建 | 显示配置创建向导 | 创建/取消 |
| 文件不存在 | 跳过或终止 | 列出缺失文件 | 跳过/重试/终止 |
| SMTP连接失败 | 重试或配置 | 显示连接错误详情 | 重试/配置/跳过 |
| 磁盘空间不足 | 终止 | 提示清理空间 | 终止 |
| 步骤执行超时 | 重试 | 显示超时信息 | 重试/跳过/终止 |

### 8.3 错误恢复机制

```typescript
interface ErrorRecovery {
  // 重试当前步骤
  retry(): Promise<void>;
  
  // 跳过当前步骤
  skip(): Promise<void>;
  
  // 回滚到指定步骤
  rollback(stepId: string): Promise<void>;
  
  // 终止整个流程
  abort(): Promise<void>;
}

class ErrorHandler {
  async handle(error: PublishError, context: PublishContext): Promise<ErrorAction> {
    // 记录错误日志
    await this.logError(error, context);
    
    // 根据错误类型选择处理策略
    switch (error.type) {
      case ErrorType.CONFIG_NOT_FOUND:
        return await this.handleConfigNotFound(error);
        
      case ErrorType.FILE_NOT_FOUND:
        return await this.handleFileNotFound(error, context);
        
      case ErrorType.SMTP_CONNECTION_FAILED:
        return await this.handleSmtpError(error);
        
      case ErrorType.STEP_TIMEOUT:
        return await this.handleTimeout(error, context);
        
      default:
        return { action: 'abort', reason: '未知错误类型' };
    }
  }
  
  private async handleFileNotFound(
    error: PublishError,
    context: PublishContext
  ): Promise<ErrorAction> {
    const missingFiles = error.details?.files as string[];
    
    const choice = await vscode.window.showErrorMessage(
      `以下文件不存在:\n${missingFiles.join('\n')}`,
      { modal: true },
      '跳过缺失文件',
      '重新检查',
      '终止发布'
    );
    
    switch (choice) {
      case '跳过缺失文件':
        return { action: 'skip', skipFiles: missingFiles };
      case '重新检查':
        return { action: 'retry' };
      default:
        return { action: 'abort' };
    }
  }
}
```

### 8.4 日志记录

```typescript
interface Logger {
  debug(message: string, meta?: Record<string, unknown>): void;
  info(message: string, meta?: Record<string, unknown>): void;
  warn(message: string, meta?: Record<string, unknown>): void;
  error(message: string, error?: Error, meta?: Record<string, unknown>): void;
}

class PublishLogger implements Logger {
  private logs: LogEntry[] = [];
  
  log(level: LogLevel, message: string, meta?: Record<string, unknown>): void {
    const entry: LogEntry = {
      timestamp: new Date(),
      level,
      message,
      details: meta
    };
    
    this.logs.push(entry);
    
    // 输出到控制台
    console.log(`[${level.toUpperCase()}] ${message}`, meta);
    
    // 输出到VS Code输出频道
    this.outputChannel.appendLine(
      `[${entry.timestamp.toISOString()}] ${level.toUpperCase()}: ${message}`
    );
  }
  
  export(): string {
    return this.logs.map(log => 
      `[${log.timestamp.toISOString()}] ${log.level.toUpperCase()}: ${log.message}`
    ).join('\n');
  }
}
```
