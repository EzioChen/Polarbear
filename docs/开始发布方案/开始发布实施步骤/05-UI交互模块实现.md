### 2.1 创建Webview面板

**文件路径**：`src/publishFlow/webview/PublishFlowPanel.ts`

**操作内容**：

```typescript
import * as vscode from 'vscode';
import * as path from 'path';
import { PublishManager } from '../PublishManager';

/**
 * 发布流程Webview面板
 */
export class PublishFlowPanel {
  public static currentPanel: PublishFlowPanel | undefined;
  private readonly panel: vscode.WebviewPanel;
  private disposables: vscode.Disposable[] = [];
  
  private constructor(
    panel: vscode.WebviewPanel,
    private context: vscode.ExtensionContext,
    private publishManager: PublishManager
  ) {
    this.panel = panel;
    
    // 设置Webview内容
    this.panel.webview.html = this.getHtmlContent();
    
    // 设置PublishManager的Webview面板
    this.publishManager.setWebviewPanel(this.panel);
    
    // 监听消息
    this.panel.webview.onDidReceiveMessage(
      async (message) => {
        switch (message.type) {
          case 'startPublish':
            await this.publishManager.startPublish();
            break;
          case 'cancelPublish':
            await this.publishManager.cancelPublish();
            break;
          case 'resetPublish':
            this.publishManager.reset();
            break;
          case 'getInitialState':
            this.sendInitialState();
            break;
        }
      },
      null,
      this.disposables
    );
    
    // 监听面板关闭
    this.panel.onDidDispose(
      () => {
        this.dispose();
      },
      null,
      this.disposables
    );
  }
  
  /**
   * 创建或显示面板
   */
  static createOrShow(
    context: vscode.ExtensionContext,
    publishManager: PublishManager
  ): PublishFlowPanel {
    const column = vscode.window.activeTextEditor
      ? vscode.window.activeTextEditor.viewColumn
      : undefined;
    
    // 如果面板已存在，则显示
    if (PublishFlowPanel.currentPanel) {
      PublishFlowPanel.currentPanel.panel.reveal(column);
      return PublishFlowPanel.currentPanel;
    }
    
    // 创建新面板
    const panel = vscode.window.createWebviewPanel(
      'publishFlow',
      '发布流程',
      column || vscode.ViewColumn.One,
      {
        enableScripts: true,
        retainContextWhenHidden: true,
        localResourceRoots: [
          vscode.Uri.file(path.join(context.extensionPath, 'webview/dist'))
        ]
      }
    );
    
    PublishFlowPanel.currentPanel = new PublishFlowPanel(
      panel,
      context,
      publishManager
    );
    
    return PublishFlowPanel.currentPanel;
  }
  
  /**
   * 发送初始状态
   */
  private sendInitialState(): void {
    this.panel.webview.postMessage({
      type: 'initialState',
      data: {
        state: this.publishManager.getState(),
        progress: this.publishManager.getProgress(),
        steps: this.publishManager.getSteps()
      }
    });
  }
  
  /**
   * 获取HTML内容
   */
  private getHtmlContent(): string {
    const webviewPath = path.join(
      this.context.extensionPath,
      'webview/dist'
    );
    
    // 读取构建后的HTML文件
    const htmlPath = path.join(webviewPath, 'index.html');
    let htmlContent = '';
    
    try {
      htmlContent = require('fs').readFileSync(htmlPath, 'utf-8');
    } catch {
      // 开发模式：返回基础HTML
      return this.getDevHtmlContent();
    }
    
    // 替换资源路径
    const webview = this.panel.webview;
    const baseUri = webview.asWebviewUri(vscode.Uri.file(webviewPath));
    
    htmlContent = htmlContent.replace(
      /(href|src)="([^"]+)"/g,
      (match, attr, value) => {
        if (value.startsWith('http') || value.startsWith('data:')) {
          return match;
        }
        return `${attr}="${baseUri}/${value.replace(/^\//, '')}"`;
      }
    );
    
    return htmlContent;
  }
  
  /**
   * 开发模式HTML
   */
  private getDevHtmlContent(): string {
    return `<!DOCTYPE html>
<html lang="zh-CN">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>发布流程</title>
  <style>
    body {
      font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
      padding: 20px;
      background: var(--vscode-editor-background);
      color: var(--vscode-editor-foreground);
    }
    .loading {
      text-align: center;
      padding: 40px;
    }
  </style>
</head>
<body>
  <div class="loading">
    <h2>发布流程</h2>
    <p>正在加载...</p>
  </div>
  <script>
    const vscode = acquireVsCodeApi();
    vscode.postMessage({ type: 'getInitialState' });
  </script>
</body>
</html>`;
  }
  
  /**
   * 释放资源
   */
  dispose(): void {
    PublishFlowPanel.currentPanel = undefined;
    this.panel.dispose();
    this.disposables.forEach(d => d.dispose());
    this.disposables = [];
  }
}
```

<template>
  <div class="publish-flow">
    <!-- 标题区域 -->

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue';

// 类型定义
interface Step {
  id: string;
  name: string;
  status: 'pending' | 'running' | 'completed' | 'failed' | 'warning';
  order: number;
  details?: Array<{ label: string; value: string }>;
  error?: string;
  completedAt?: string;
}

interface ConfigCheck {
  name: string;
  valid: boolean;
  error?: string;
}

interface PublishError {
  type: string;
  message: string;
  recoverable: boolean;
}

// 状态
const state = ref('IDLE');
const progress = ref(0);
const steps = ref<Step[]>([]);
const configChecks = ref<ConfigCheck[]>([]);
const error = ref<PublishError | null>(null);
const isLoading = ref(false);

// 计算属性
const isRunning = computed(() => 
  ['CHECKING', 'PACKAGING', 'GENERATING', 'ASSEMBLING'].includes(state.value)
);

const canStart = computed(() => 
  state.value === 'IDLE' || state.value === 'ERROR'
);

const canCancel = computed(() => isRunning.value);

const canReset = computed(() => 
  state.value === 'COMPLETED' || state.value === 'ERROR'
);

const stateClass = computed(() => {
  const classes: Record<string, string> = {
    'IDLE': 'state-idle',
    'CHECKING': 'state-running',
    'PACKAGING': 'state-running',
    'GENERATING': 'state-running',
    'ASSEMBLING': 'state-running',
    'COMPLETED': 'state-completed',
    'ERROR': 'state-error'
  };
  return classes[state.value] || 'state-idle';
});

const stateText = computed(() => {
  const texts: Record<string, string> = {
    'IDLE': '准备就绪',
    'CHECKING': '检查配置中...',
    'PACKAGING': '打包文件中...',
    'GENERATING': '生成文档中...',
    'ASSEMBLING': '组装邮件中...',
    'COMPLETED': '发布完成',
    'ERROR': '发生错误'
  };
  return texts[state.value] || '未知状态';
});

// VS Code API
const vscode = (window as any).acquireVsCodeApi();

// 方法
const startPublish = () => {
  isLoading.value = true;
  vscode.postMessage({ type: 'startPublish' });
};

const cancelPublish = () => {
  vscode.postMessage({ type: 'cancelPublish' });
};

const resetPublish = () => {
  vscode.postMessage({ type: 'resetPublish' });
  error.value = null;
  configChecks.value = [];
};

const clearError = () => {
  error.value = null;
};

const formatTime = (timeStr: string): string => {
  const date = new Date(timeStr);
  return date.toLocaleTimeString('zh-CN', { 
    hour: '2-digit', 
    minute: '2-digit',
    second: '2-digit'
  });
};

// 消息处理
const handleMessage = (event: MessageEvent) => {
  const message = event.data;
  
  switch (message.type) {
    case 'initialState':
      state.value = message.data.state;
      progress.value = message.data.progress;
      steps.value = message.data.steps;
      break;
      
    case 'stateUpdate':
      state.value = message.data.state;
      progress.value = message.data.progress;
      isLoading.value = false;
      break;
      
    case 'stepUpdate':
      const stepIndex = steps.value.findIndex(s => s.id === message.data.id);
      if (stepIndex >= 0) {
        steps.value[stepIndex] = message.data;
      }
      break;
      
    case 'configCheck':
      configChecks.value = message.data.checks;
      break;
      
    case 'error':
      error.value = message.data;
      isLoading.value = false;
      break;
      
    case 'completed':
      isLoading.value = false;
      break;
  }
};

// 生命周期
onMounted(() => {
  window.addEventListener('message', handleMessage);
  vscode.postMessage({ type: 'getInitialState' });
});
</script>
