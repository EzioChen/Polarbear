/**
 * 优化的文件打包方法
 */
private async createZipOptimized(
  files: FileItem[],
  outputPath: string,
  onProgress?: (progress: number) => void
): Promise<PackResult> {
  return new Promise((resolve, reject) => {
    const output = createWriteStream(outputPath);
    const archive = archiver('zip', {
      zlib: { level: 6 }, // 平衡压缩率和速度
      store: false
    });
    
    let processedSize = 0;
    const totalSize = files.reduce((sum, f) => sum + f.size, 0);
    
    // 批量添加文件以提高性能
    const batchSize = 10;
    let currentBatch = 0;
    
    archive.on('progress', (progress) => {
      const percent = Math.round((progress.fs.processedBytes / totalSize) * 100);
      onProgress?.(percent);
    });
    
    archive.on('error', (err) => {
      reject(new PublishError(
        ErrorType.STEP_EXECUTION_FAILED,
        `ZIP压缩失败: ${err.message}`,
        {},
        true
      ));
    });
    
    archive.on('close', () => {
      resolve({
        zipPath: outputPath,
        manifestPath: '',
        fileCount: files.length,
        totalSize: archive.pointer()
      });
    });
    
    archive.pipe(output);
    
    // 分批添加文件
    const addFilesInBatches = async () => {
      for (let i = 0; i < files.length; i += batchSize) {
        const batch = files.slice(i, i + batchSize);
        
        for (const file of batch) {
          archive.file(file.sourcePath, { name: file.targetPath });
        }
        
        // 让出控制权，避免阻塞
        await new Promise(resolve => setImmediate(resolve));
      }
      
      await archive.finalize();
    };
    
    addFilesInBatches().catch(reject);
  });
}

/**
 * 优化的PDF生成方法
 */
private async generatePDFOptimized(
  markdownContent: string,
  flowId: string,
  version: string
): Promise<string> {
  // 使用浏览器池避免重复启动
  const browser = await this.getBrowserFromPool();
  
  try {
    const page = await browser.newPage();
    
    // 设置内容
    const htmlContent = this.generateHTML(markdownContent, version);
    await page.setContent(htmlContent, { waitUntil: 'networkidle0' });
    
    // 生成PDF
    const pdfPath = this.cacheStore.getPath('release-notes.pdf', flowId);
    await page.pdf({
      path: pdfPath,
      format: 'A4',
      margin: { top: '2cm', right: '2cm', bottom: '2cm', left: '2cm' },
      printBackground: true,
      // 优化：禁用不必要的功能
      displayHeaderFooter: false,
      preferCSSPageSize: true
    });
    
    return pdfPath;
  } finally {
    await this.releaseBrowserToPool(browser);
  }
}

/**
 * 浏览器池
 */
private browserPool: any[] = [];
private maxPoolSize = 2;

private async getBrowserFromPool(): Promise<any> {
  if (this.browserPool.length > 0) {
    return this.browserPool.pop();
  }
  
  return puppeteer.launch({
    headless: true,
    args: [
      '--no-sandbox',
      '--disable-setuid-sandbox',
      '--disable-dev-shm-usage',
      '--disable-accelerated-2d-canvas',
      '--disable-gpu'
    ]
  });
}

private async releaseBrowserToPool(browser: any): Promise<void> {
  if (this.browserPool.length < this.maxPoolSize) {
    this.browserPool.push(browser);
  } else {
    await browser.close();
  }
}

/**

private async cleanup(): Promise<void> {
  // 清理缓存文件
  const cacheDir = path.join(this.context.extensionPath, '.releasePlan', 'cache');
  
  if (fs.existsSync(cacheDir)) {
    // 只保留最近5个流程的缓存
    const flowDirs = fs.readdirSync(cacheDir)
      .filter(dir => dir.startsWith('flow-'))
      .map(dir => ({
        name: dir,
        time: fs.statSync(path.join(cacheDir, dir)).mtime.getTime()
      }))
      .sort((a, b) => b.time - a.time);
    
    // 删除旧的缓存
    for (let i = 5; i < flowDirs.length; i++) {
      const dirPath = path.join(cacheDir, flowDirs[i].name);
      fs.rmSync(dirPath, { recursive: true, force: true });
    }
  }
  
  // 强制垃圾回收（如果可用）
  if (global.gc) {
    global.gc();
  }
}

 * 错误处理帮助类

export class ErrorHelper {
  /**
   * 分析错误类型
   */
  static categorizeError(error: Error): ErrorCategory {
    const message = error.message.toLowerCase();
    
    if (message.includes('config') || message.includes('配置')) {
      return ErrorCategory.CONFIG;
    }
    if (message.includes('network') || message.includes('connect') || message.includes('timeout')) {
      return ErrorCategory.NETWORK;

    if (message.includes('enoent') || message.includes('permission') || message.includes('eacces')) {
      return ErrorCategory.FILE_SYSTEM;

    if (message.includes('validation') || message.includes('invalid')) {
      return ErrorCategory.VALIDATION;

    return ErrorCategory.UNKNOWN;

// src/publishFlow/Logger.ts

/**
 * 结构化日志记录器
 */
export class StructuredLogger {
  private logEntries: LogEntry[] = [];
  private maxEntries = 1000;
  
  constructor(private context: string) {}
  
  /**
   * 记录信息日志
   */
  info(message: string, metadata?: Record<string, any>): void {
    this.log('INFO', message, metadata);
  }
  
  /**
   * 记录警告日志
   */
  warn(message: string, metadata?: Record<string, any>): void {
    this.log('WARN', message, metadata);
  }
  
  /**
   * 记录错误日志
   */
  error(message: string, error?: Error, metadata?: Record<string, any>): void {
    this.log('ERROR', message, {
      ...metadata,
      error: error ? {
        message: error.message,
        stack: error.stack,
        name: error.name
      } : undefined
    });
  }
  
  /**
   * 记录调试日志
   */
  debug(message: string, metadata?: Record<string, any>): void {
    this.log('DEBUG', message, metadata);
  }
  
  /**
   * 获取日志历史
   */
  getHistory(): LogEntry[] {
    return [...this.logEntries];
  }
  
  /**
   * 导出日志
   */
  export(): string {
    return JSON.stringify(this.logEntries, null, 2);
  }
  
  private log(level: LogLevel, message: string, metadata?: Record<string, any>): void {
    const entry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      context: this.context,
      message,
      metadata
    };
    
    this.logEntries.push(entry);
    
    // 限制日志条目数量
    if (this.logEntries.length > this.maxEntries) {
      this.logEntries.shift();
    }
    
    // 同时输出到控制台
    const consoleMethod = level === 'ERROR' ? console.error :
                          level === 'WARN' ? console.warn :
                          level === 'DEBUG' ? console.debug : console.log;
    
    consoleMethod(`[${level}] ${this.context}: ${message}`, metadata || '');
  }
}

interface LogEntry {
  timestamp: string;
  level: LogLevel;
  context: string;
  message: string;
  metadata?: Record<string, any>;
}

type LogLevel = 'INFO' | 'WARN' | 'ERROR' | 'DEBUG';
```

2. 点击"开始发布"
3. 等待流程完成
4. 检查邮件内容
5. 发送邮件

## 故障排除

### 配置检查失败

检查配置文件是否存在且格式正确。

### 文件打包失败

检查文件路径是否正确，文件是否存在。

### 邮件发送失败

检查SMTP配置和网络连接。
```
