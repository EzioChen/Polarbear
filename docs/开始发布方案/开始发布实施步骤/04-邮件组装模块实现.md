### 2.1 实现邮件组装模块

**文件路径**：`src/publishFlow/modules/EmailAsmModule.ts`

**操作内容**：

```typescript
import * as vscode from 'vscode';
import * as path from 'path';
import * as fs from 'fs-extra';
import { CacheStore } from '../CacheStore';
import { PublishLogger, logger } from '../logger';
import { PublishError, ErrorType } from '../errors';
import { EmailOutput, DocumentationOutput, PackagingOutput } from '../types';

/**
 * 发布说明配置
 */
interface ReleaseNotesConfig {
  version: string;
  createdAt: string;
  updatedAt: string;
  versionLevel: 'A' | 'B' | 'C';
  releaseNotes: string;
  chipRange: string[];
  sdkVersions: Array<{
    id: string;
    chip: string;
    version: string;
    remark: string;
  }>;
  sdkVersion: string;
  appVersion: string;
  remarks: string;
  metadata: {
    createdBy: string;
    author: string;
  };
}

/**
 * SMTP配置
 */
interface SMTPConfig {
  host: string;
  port: number;
  secure: boolean;
  auth: {
    user: string;
    pass: string;
  };
  tls?: {
    rejectUnauthorized: boolean;
  };
  defaultTo?: Array<{ name?: string; email: string }>;
  defaultCc?: Array<{ name?: string; email: string }>;
}

/**
 * 邮件组装模块
 */
export class EmailAsmModule {
  private configPath: string;
  private smtpConfigPath: string;
  private workspacePath: string;
  
  constructor(
    private cacheStore: CacheStore,
    private logger: PublishLogger
  ) {
    this.workspacePath = this.getWorkspacePath();
    this.configPath = path.join(
      this.workspacePath,
      '.releasePlan',
      'release-notes.json'
    );
    this.smtpConfigPath = path.join(
      this.workspacePath,
      '.releasePlan',
      'smtp-config.json'
    );
  }
  
  /**
   * 执行邮件组装流程
   */
  async execute(
    flowId: string,
    packaging: PackagingOutput,
    documentation: DocumentationOutput
  ): Promise<EmailOutput> {
    this.logger.info('开始邮件组装流程', { flowId });
    
    try {
      // 1. 检查SMTP配置
      await this.checkSMTPConfig();
      
      // 2. 读取发布说明
      const releaseConfig = await this.loadReleaseConfig();
      
      // 3. 生成邮件内容
      const emailContent = this.generateEmailContent(releaseConfig);
      
      // 4. 组装附件列表
      const attachments = [
        packaging.zipPath,
        documentation.pdfPath
      ].filter(Boolean);
      
      this.logger.info('邮件组装完成', { 
        flowId,
        subject: emailContent.subject,
        attachmentCount: attachments.length
      });
      
      return {
        subject: emailContent.subject,
        body: emailContent.body,
        attachments
      };
      
    } catch (error) {
      this.logger.error('邮件组装失败', error as Error, { flowId });
      throw error;
    }
  }
  
  /**
   * 检查SMTP配置
   */
  private async checkSMTPConfig(): Promise<void> {
    // 检查配置文件存在性
    if (!(await fs.pathExists(this.smtpConfigPath))) {
      throw new PublishError(
        ErrorType.CONFIG_NOT_FOUND,
        `SMTP配置文件不存在: ${this.smtpConfigPath}`,
        { configPath: this.smtpConfigPath },
        true
      );
    }
    
    // 读取并验证配置
    let smtpConfig: SMTPConfig;
    try {
      const content = await fs.readFile(this.smtpConfigPath, 'utf-8');
      smtpConfig = JSON.parse(content);
    } catch (error) {
      throw new PublishError(
        ErrorType.CONFIG_INVALID,
        `SMTP配置解析失败: ${(error as Error).message}`,
        {},
        true
      );
    }
    
    // 验证必填字段
    const requiredFields = ['host', 'port', 'auth.user'];
    const missingFields = requiredFields.filter(field => {
      const parts = field.split('.');
      let value: unknown = smtpConfig;
      for (const part of parts) {
        value = (value as Record<string, unknown>)?.[part];
      }
      return !value;
    });
    
    if (missingFields.length > 0) {
      throw new PublishError(
        ErrorType.CONFIG_INCOMPLETE,
        `SMTP配置不完整，缺少: ${missingFields.join(', ')}`,
        { missingFields },
        true
      );
    }
  }
  
  /**
   * 测试SMTP连接
   */
  async testSMTPConnection(): Promise<{
    success: boolean;
    message: string;
  }> {
    try {
      await this.checkSMTPConfig();
      
      const content = await fs.readFile(this.smtpConfigPath, 'utf-8');
      const config: SMTPConfig = JSON.parse(content);
      
      // 动态导入 nodemailer
      const nodemailer = await import('nodemailer');
      
      const transporter = nodemailer.createTransport({
        host: config.host,
        port: config.port,
        secure: config.secure,
        auth: {
          user: config.auth.user,
          pass: config.auth.pass
        },
        tls: config.tls
      });
      
      // 验证连接
      await transporter.verify();
      
      return {
        success: true,
        message: 'SMTP连接测试成功'
      };
    } catch (error) {
      return {
        success: false,
        message: `SMTP连接测试失败: ${(error as Error).message}`
      };
    }
  }
  
  /**
   * 加载发布说明配置
   */
  private async loadReleaseConfig(): Promise<ReleaseNotesConfig> {
    const content = await fs.readFile(this.configPath, 'utf-8');
    return JSON.parse(content);
  }
  
  /**
   * 生成邮件内容
   */
  private generateEmailContent(config: ReleaseNotesConfig): {
    subject: string;
    body: string;
  } {
    // 生成主题
    const subject = `[发布] ${config.appVersion} - ${config.versionLevel}级版本发布`;
    
    // 生成正文
    const versionLevelText = {
      'A': '重大版本',
      'B': '次要版本',
      'C': '补丁版本'
    };
    
    const chipRangeText = config.chipRange.join(', ');
    
    const sdkVersionsText = config.sdkVersions.map(v => 
      `- ${v.chip}: ${v.version}${v.remark ? ` (${v.remark})` : ''}`
    ).join('\n');
    
    const body = `各位同事，您好！

${config.appVersion} 版本已发布，详细信息如下：

**版本信息**
- 应用版本：${config.appVersion}
- SDK版本：${config.sdkVersion}
- 版本级别：${versionLevelText[config.versionLevel]}
- 适用芯片：${chipRangeText}

**SDK版本详情**
${sdkVersionsText}

**发布说明**
${config.releaseNotes}

**附件说明**
1. package.zip - 发布包文件
2. release-notes.pdf - 发布说明文档

如有问题，请及时反馈。

谢谢！

---
此邮件由 PolarBear Release System 自动生成
发送时间：${new Date().toLocaleString('zh-CN')}`;
    
    return { subject, body };
  }
  
  /**
   * 获取默认收件人
   */
  async getDefaultRecipients(): Promise<{
    to: string[];
    cc: string[];
  }> {
    try {
      if (!(await fs.pathExists(this.smtpConfigPath))) {
        return { to: [], cc: [] };
      }
      
      const content = await fs.readFile(this.smtpConfigPath, 'utf-8');
      const config: SMTPConfig = JSON.parse(content);
      
      return {
        to: (config.defaultTo || []).map(c => c.email),
        cc: (config.defaultCc || []).map(c => c.email)
      };
    } catch {
      return { to: [], cc: [] };
    }
  }
  
  /**
   * 获取工作区路径
   */
  private getWorkspacePath(): string {
    if (vscode.workspace.workspaceFolders && vscode.workspace.workspaceFolders.length > 0) {
      return vscode.workspace.workspaceFolders[0].uri.fsPath;
    }
    throw new Error('没有打开的工作区');
  }
}
```

### 2.2 更新 PublishManager 集成邮件组装

**操作内容**：

在 `PublishManager` 类中添加 `EmailAsmModule` 集成：

```typescript
import { EmailAsmModule } from './modules/EmailAsmModule';

// 在类属性中添加
private emailAsmModule: EmailAsmModule;

// 在构造函数中初始化
constructor(context: vscode.ExtensionContext) {
  // ... 现有代码 ...
  this.emailAsmModule = new EmailAsmModule(this.cacheStore, this.logger);
  // ... 现有代码 ...
}

// 在 startPublish 方法中启用步骤4
async startPublish(): Promise<void> {
  // ... 现有代码 ...
  
  // 步骤3: 文档生成
  const documentationOutput = await this.executeStep('generate', async () => {
    return await this.docGenModule.execute(flowId);
  });
  context.documentation = documentationOutput;
  
  // 步骤4: 邮件组装
  const emailOutput = await this.executeStep('assemble', async () => {
    return await this.emailAsmModule.execute(
      flowId,
      packagingOutput,
      documentationOutput
    );
  });
  context.email = emailOutput;
  
  // 步骤5: 打开邮件编辑器
  await this.openEmailEditor(emailOutput);
  
  // ... 完成流程 ...
}

/**
 * 打开邮件编辑器
 */
private async openEmailEditor(emailOutput: EmailOutput): Promise<void> {
  // 获取默认收件人
  const defaultRecipients = await this.emailAsmModule.getDefaultRecipients();
  
  // 触发打开邮件编辑器命令
  await vscode.commands.executeCommand('polarbear.email.openEditor', {
    subject: emailOutput.subject,
    body: emailOutput.body,
    attachments: emailOutput.attachments,
    defaultTo: defaultRecipients.to,
    defaultCc: defaultRecipients.cc
  });
}
```

```typescript
/**
 * 打开邮件编辑器（带预填充内容）
 */
static async createWithContent(
  context: vscode.ExtensionContext,
  emailService: EmailService,
  content: {
    subject?: string;
    body?: string;
    attachments?: string[];
    defaultTo?: string[];
    defaultCc?: string[];
  }
): Promise<EmailEditorPanel> {
  const panel = await this.create(context, emailService);
  
  // 预填充内容
  panel.webviewPanel.webview.postMessage({
    type: 'prefill',
    data: content
  });
  
  return panel;
}
```

**文件路径**：`webview/src/views/EmailEditorView.vue`

**操作内容**：

添加预填充消息处理：

```typescript
case 'prefill':
  // 预填充邮件内容
  if (message.data) {
    if (message.data.subject) {
      form.value.subject = message.data.subject;
    }
    if (message.data.body) {
      form.value.content = message.data.body;
    }
    if (message.data.attachments) {
      attachments.value = message.data.attachments.map((path: string) => ({
        path,
        name: path.split('/').pop() || path
      }));
    }
    if (message.data.defaultTo && message.data.defaultTo.length > 0) {
      form.value.to = message.data.defaultTo.join(';');
    }
    if (message.data.defaultCc && message.data.defaultCc.length > 0) {
      form.value.cc = message.data.defaultCc.join(';');
    }
  }
  break;
```
