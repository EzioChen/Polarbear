```typescript
import * as vscode from 'vscode';

export type LogLevel = 'debug' | 'info' | 'warn' | 'error';

export interface LogEntry {
  timestamp: Date;
  level: LogLevel;
  message: string;
  details?: Record<string, unknown>;
}

export class PublishLogger {
  private logs: LogEntry[] = [];
  private outputChannel: vscode.OutputChannel;
  
  constructor(channelName: string = 'PolarBear Publish') {
    this.outputChannel = vscode.window.createOutputChannel(channelName);
  }
  
  /**
   * 记录调试日志
   */
  debug(message: string, details?: Record<string, unknown>): void {
    this.log('debug', message, details);
  }
  
  /**
   * 记录信息日志
   */
  info(message: string, details?: Record<string, unknown>): void {
    this.log('info', message, details);
  }
  
  /**
   * 记录警告日志
   */
  warn(message: string, details?: Record<string, unknown>): void {
    this.log('warn', message, details);
  }
  
  /**
   * 记录错误日志
   */
  error(message: string, error?: Error, details?: Record<string, unknown>): void {
    this.log('error', message, {
      ...details,
      error: error?.message,
      stack: error?.stack
    });
  }
  
  /**
   * 内部日志记录方法
   */
  private log(level: LogLevel, message: string, details?: Record<string, unknown>): void {
    const entry: LogEntry = {
      timestamp: new Date(),
      level,
      message,
      details
    };
    
    this.logs.push(entry);
    
    // 输出到控制台
    console.log(`[${level.toUpperCase()}] ${message}`, details);
    
    // 输出到VS Code输出频道
    const timestamp = entry.timestamp.toISOString();
    this.outputChannel.appendLine(`[${timestamp}] ${level.toUpperCase()}: ${message}`);
    
    if (details) {
      this.outputChannel.appendLine(`  Details: ${JSON.stringify(details, null, 2)}`);
    }
  }
  
  /**
   * 导出日志
   */
  export(): string {
    return this.logs.map(log => {
      const base = `[${log.timestamp.toISOString()}] ${log.level.toUpperCase()}: ${log.message}`;
      if (log.details) {
        return `${base}\n  Details: ${JSON.stringify(log.details, null, 2)}`;
      }
      return base;
    }).join('\n');
  }
  
  /**
   * 清空日志
   */
  clear(): void {
    this.logs = [];
    this.outputChannel.clear();
  }
  
  /**
   * 显示输出频道
   */
  show(): void {
    this.outputChannel.show();
  }
  
  /**
   * 获取所有日志
   */
  getLogs(): LogEntry[] {
    return [...this.logs];
  }
}

// 导出单例
export const logger = new PublishLogger();
```

### 2.5 实现缓存存储模块

**文件路径**：`src/publishFlow/CacheStore.ts`

**操作内容**：

```typescript
import * as vscode from 'vscode';
import * as path from 'path';
import * as fs from 'fs-extra';
import { CacheFile } from './types';
import { logger } from './logger';

/**
 * 缓存存储管理器
 */
export class CacheStore {
  private cacheDir: string;
  private maxAge: number;  // 缓存最大保留时间（毫秒）
  
  constructor(
    private context: vscode.ExtensionContext,
    maxAgeHours: number = 24
  ) {
    this.cacheDir = path.join(context.extensionPath, '.cache', 'publish');
    this.maxAge = maxAgeHours * 60 * 60 * 1000;
    
    // 确保缓存目录存在
    this.ensureCacheDir();
    
    // 启动时清理过期缓存
    this.cleanupExpired();
  }
  
  /**
   * 确保缓存目录存在
   */
  private async ensureCacheDir(): Promise<void> {
    await fs.ensureDir(this.cacheDir);
  }
  
  /**
   * 创建流程缓存目录
   */
  async createFlowDir(flowId: string): Promise<string> {
    const flowDir = path.join(this.cacheDir, flowId);
    await fs.ensureDir(flowDir);
    
    // 创建子目录
    await fs.ensureDir(path.join(flowDir, 'docs'));
    await fs.ensureDir(path.join(flowDir, 'logs'));
    
    logger.debug(`创建流程缓存目录: ${flowDir}`, { flowId });
    return flowDir;
  }
  
  /**
   * 保存文件到缓存
   */
  async save(
    sourcePath: string,
    filename: string,
    flowId: string,
    type: CacheFile['type']
  ): Promise<string> {
    const flowDir = path.join(this.cacheDir, flowId);
    const targetPath = path.join(flowDir, filename);
    
    // 复制文件
    await fs.copy(sourcePath, targetPath);
    
    const stats = await fs.stat(targetPath);
    
    logger.info(`文件已保存到缓存`, {
      flowId,
      filename,
      type,
      size: stats.size
    });
    
    return targetPath;
  }
  
  /**
   * 保存Buffer到缓存
   */
  async saveBuffer(
    buffer: Buffer,
    filename: string,
    flowId: string,
    type: CacheFile['type']
  ): Promise<string> {
    const flowDir = path.join(this.cacheDir, flowId);
    const targetPath = path.join(flowDir, filename);
    
    await fs.writeFile(targetPath, buffer);
    
    logger.info(`Buffer已保存到缓存`, {
      flowId,
      filename,
      type,
      size: buffer.length
    });
    
    return targetPath;
  }
  
  /**
   * 获取缓存文件路径
   */
  getPath(filename: string, flowId: string): string {
    return path.join(this.cacheDir, flowId, filename);
  }
  
  /**
   * 读取缓存文件
   */
  async read(filename: string, flowId: string): Promise<Buffer> {
    const filePath = this.getPath(filename, flowId);
    return fs.readFile(filePath);
  }
  
  /**
   * 检查文件是否存在
   */
  async exists(filename: string, flowId: string): Promise<boolean> {
    const filePath = this.getPath(filename, flowId);
    return fs.pathExists(filePath);
  }
  
  /**
   * 列出流程的所有缓存文件
   */
  async list(flowId: string): Promise<CacheFile[]> {
    const flowDir = path.join(this.cacheDir, flowId);
    
    if (!(await fs.pathExists(flowDir))) {
      return [];
    }
    
    const files: CacheFile[] = [];
    const entries = await fs.readdir(flowDir, { withFileTypes: true });
    
    for (const entry of entries) {
      if (entry.isFile()) {
        const filePath = path.join(flowDir, entry.name);
        const stats = await fs.stat(filePath);
        
        files.push({
          id: `${flowId}_${entry.name}`,
          flowId,
          type: this.inferFileType(entry.name),
          filename: entry.name,
          path: filePath,
          size: stats.size,
          createdAt: stats.birthtime,
          expiresAt: new Date(stats.birthtime.getTime() + this.maxAge)
        });
      }
    }
    
    return files;
  }
  
  /**
   * 推断文件类型
   */
  private inferFileType(filename: string): CacheFile['type'] {
    const ext = path.extname(filename).toLowerCase();
    switch (ext) {
      case '.zip':
        return 'zip';
      case '.pdf':
        return 'pdf';
      case '.md':
      case '.markdown':
        return 'md';
      case '.log':
        return 'log';
      default:
        return 'log';
    }
  }
  
  /**
   * 清理指定流程的缓存
   */
  async cleanup(flowId?: string): Promise<void> {
    if (flowId) {
      const flowDir = path.join(this.cacheDir, flowId);
      if (await fs.pathExists(flowDir)) {
        await fs.remove(flowDir);
        logger.info(`流程缓存已清理`, { flowId });
      }
    } else {
      // 清理所有缓存
      if (await fs.pathExists(this.cacheDir)) {
        await fs.emptyDir(this.cacheDir);
        logger.info('所有缓存已清理');
      }
    }
  }
  
  /**
   * 清理过期缓存
   */
  private async cleanupExpired(): Promise<void> {
    if (!(await fs.pathExists(this.cacheDir))) {
      return;
    }
    
    const now = Date.now();
    const entries = await fs.readdir(this.cacheDir, { withFileTypes: true });
    
    for (const entry of entries) {
      if (entry.isDirectory()) {
        const flowDir = path.join(this.cacheDir, entry.name);
        const stats = await fs.stat(flowDir);
        const age = now - stats.mtime.getTime();
        
        if (age > this.maxAge) {
          await fs.remove(flowDir);
          logger.debug(`过期缓存已清理`, { flowId: entry.name, age: `${Math.round(age / 3600000)}h` });
        }
      }
    }
  }
  
  /**
   * 获取缓存目录大小
   */
  async getCacheSize(): Promise<number> {
    if (!(await fs.pathExists(this.cacheDir))) {
      return 0;
    }
    
    let totalSize = 0;
    
    const calculateSize = async (dir: string): Promise<void> => {
      const entries = await fs.readdir(dir, { withFileTypes: true });
      
      for (const entry of entries) {
        const fullPath = path.join(dir, entry.name);
        if (entry.isDirectory()) {
          await calculateSize(fullPath);
        } else {
          const stats = await fs.stat(fullPath);
          totalSize += stats.size;
        }
      }
    };
    
    await calculateSize(this.cacheDir);
    return totalSize;
  }
}
```

### 2.6 实现状态机

**文件路径**：`src/publishFlow/StateMachine.ts`

**操作内容**：

```typescript
import { PublishState, PublishStep, StepStatus } from './types';
import { EventEmitter } from 'events';

/**
 * 状态转换定义
 */
interface StateTransition {
  from: PublishState;
  to: PublishState;
  action: string;
}

/**
 * 发布流程状态机
 */
export class PublishStateMachine extends EventEmitter {
  private currentState: PublishState = 'IDLE';
  private steps: Map<string, PublishStep> = new Map();
  
  // 有效状态转换表
  private validTransitions: StateTransition[] = [
    { from: 'IDLE', to: 'CHECKING', action: 'start' },
    { from: 'CHECKING', to: 'PACKAGING', action: 'checkPassed' },
    { from: 'CHECKING', to: 'ERROR', action: 'checkFailed' },
    { from: 'PACKAGING', to: 'GENERATING', action: 'packSuccess' },
    { from: 'PACKAGING', to: 'ERROR', action: 'packFailed' },
    { from: 'GENERATING', to: 'ASSEMBLING', action: 'generateSuccess' },
    { from: 'GENERATING', to: 'ERROR', action: 'generateFailed' },
    { from: 'ASSEMBLING', to: 'COMPLETED', action: 'assembleSuccess' },
    { from: 'ASSEMBLING', to: 'ERROR', action: 'assembleFailed' },
    { from: 'ERROR', to: 'CHECKING', action: 'retry' },
    { from: 'ERROR', to: 'IDLE', action: 'reset' },
    { from: 'COMPLETED', to: 'IDLE', action: 'reset' }
  ];
  
  /**
   * 获取当前状态
   */
  getState(): PublishState {
    return this.currentState;
  }
  
  /**
   * 检查状态转换是否有效
   */
  canTransition(to: PublishState): boolean {
    return this.validTransitions.some(
      t => t.from === this.currentState && t.to === to
    );
  }
  
  /**
   * 执行状态转换
   */
  transition(to: PublishState, data?: unknown): boolean {
    if (!this.canTransition(to)) {
      throw new Error(
        `无效的状态转换: ${this.currentState} -> ${to}`
      );
    }
    
    const from = this.currentState;
    this.currentState = to;
    
    this.emit('stateChanged', {
      from,
      to,
      data
    });
    
    return true;
  }
  
  /**
   * 注册步骤
   */
  registerStep(step: PublishStep): void {
    this.steps.set(step.id, step);
  }
  
  /**
   * 获取步骤
   */
  getStep(id: string): PublishStep | undefined {
    return this.steps.get(id);
  }
  
  /**
   * 获取所有步骤
   */
  getAllSteps(): PublishStep[] {
    return Array.from(this.steps.values()).sort((a, b) => a.order - b.order);
  }
  
  /**
   * 更新步骤状态
   */
  updateStepStatus(stepId: string, status: StepStatus, error?: string): void {
    const step = this.steps.get(stepId);
    if (!step) {
      throw new Error(`步骤不存在: ${stepId}`);
    }
    
    step.status = status;
    
    if (status === 'running' && !step.startedAt) {
      step.startedAt = new Date();
    }
    
    if (status === 'completed' || status === 'failed') {
      step.completedAt = new Date();
    }
    
    if (error) {
      step.error = error;
    }
    
    this.emit('stepUpdated', step);
  }
  
  /**
   * 更新步骤详情
   */
  updateStepDetails(stepId: string, details: PublishStep['details']): void {
    const step = this.steps.get(stepId);
    if (step) {
      step.details = details;
      this.emit('stepUpdated', step);
    }
  }
  
  /**
   * 重置状态机
   */
  reset(): void {
    this.currentState = 'IDLE';
    this.steps.clear();
    this.emit('reset');
  }
  
  /**
   * 计算进度百分比
   */
  getProgress(): number {
    const steps = this.getAllSteps();
    if (steps.length === 0) return 0;
    
    const completed = steps.filter(s => s.status === 'completed').length;
    return Math.round((completed / steps.length) * 100);
  }
  
  /**
   * 获取当前执行的步骤
   */
  getCurrentStep(): PublishStep | undefined {
    return this.getAllSteps().find(s => s.status === 'running');
  }
  
  /**
   * 检查是否可以开始
   */
  canStart(): boolean {
    return this.currentState === 'IDLE' || this.currentState === 'ERROR';
  }
  
  /**
   * 检查是否可以取消
   */
  canCancel(): boolean {
    return ['CHECKING', 'PACKAGING', 'GENERATING', 'ASSEMBLING'].includes(
      this.currentState
    );
  }
}
```

```typescript
import * as vscode from 'vscode';
import { PublishStateMachine } from './StateMachine';
import { CacheStore } from './CacheStore';
import { PublishLogger, logger } from './logger';
import { PublishError, ConfigCheckError, ErrorType } from './errors';
import { 
  PublishState, 
  PublishStep, 
  PublishContext,
  ConfigCheckResult 
} from './types';
import * as path from 'path';
import * as fs from 'fs-extra';

/**
 * 发布流程管理器
 */
export class PublishManager {
  private stateMachine: PublishStateMachine;
  private cacheStore: CacheStore;
  private logger: PublishLogger;
  private context: vscode.ExtensionContext;
  private webviewPanel?: vscode.WebviewPanel;
  
  constructor(context: vscode.ExtensionContext) {
    this.context = context;
    this.stateMachine = new PublishStateMachine();
    this.cacheStore = new CacheStore(context);
    this.logger = new PublishLogger();
    
    this.setupEventListeners();
    this.initializeSteps();
  }
  
  /**
   * 设置事件监听
   */
  private setupEventListeners(): void {
    // 状态变更监听
    this.stateMachine.on('stateChanged', (event) => {
      this.logger.info(`状态变更: ${event.from} -> ${event.to}`);
      this.notifyWebview('stateUpdate', {
        state: event.to,
        progress: this.stateMachine.getProgress()
      });
    });
    
    // 步骤更新监听
    this.stateMachine.on('stepUpdated', (step) => {
      this.notifyWebview('stepUpdate', step);
    });
  }
  
  /**
   * 初始化步骤
   */
  private initializeSteps(): void {
    const steps: PublishStep[] = [
      {
        id: 'check',
        name: '配置检查',
        status: 'pending',
        order: 1
      },
      {
        id: 'pack',
        name: '文件打包',
        status: 'pending',
        order: 2
      },
      {
        id: 'generate',
        name: '文档生成',
        status: 'pending',
        order: 3
      },
      {
        id: 'assemble',
        name: '邮件组装',
        status: 'pending',
        order: 4
      }
    ];
    
    steps.forEach(step => this.stateMachine.registerStep(step));
  }
  
  /**
   * 启动发布流程
   */
  async startPublish(): Promise<void> {
    if (!this.stateMachine.canStart()) {
      throw new Error('发布流程正在进行中，无法启动');
    }
    
    const flowId = this.generateFlowId();
    this.logger.info('开始发布流程', { flowId });
    
    try {
      // 创建流程上下文
      const context: PublishContext = {
        flowId,
        version: '1.0.0',
        startTime: new Date(),
        workspacePath: this.getWorkspacePath()
      };
      
      // 创建缓存目录
      await this.cacheStore.createFlowDir(flowId);
      
      // 步骤1: 配置检查
      await this.executeStep('check', async () => {
        return await this.checkConfigs();
      });
      
      // 步骤2: 文件打包（在下一步实现）
      // await this.executeStep('pack', async () => { ... });
      
      // 步骤3: 文档生成（在下一步实现）
      // await this.executeStep('generate', async () => { ... });
      
      // 步骤4: 邮件组装（在下一步实现）
      // await this.executeStep('assemble', async () => { ... });
      
      this.stateMachine.transition('COMPLETED');
      this.notifyWebview('completed', { flowId });
      
    } catch (error) {
      this.handleError(error);
    }
  }
  
  /**
   * 执行单个步骤
   */
  private async executeStep<T>(
    stepId: string,
    executor: () => Promise<T>
  ): Promise<T> {
    this.stateMachine.updateStepStatus(stepId, 'running');
    this.logger.info(`开始执行步骤: ${stepId}`);
    
    try {
      const result = await executor();
      this.stateMachine.updateStepStatus(stepId, 'completed');
      this.logger.info(`步骤完成: ${stepId}`);
      return result;
    } catch (error) {
      this.stateMachine.updateStepStatus(stepId, 'failed', (error as Error).message);
      this.logger.error(`步骤失败: ${stepId}`, error as Error);
      throw error;
    }
  }
  
  /**
   * 检查配置文件
   */
  private async checkConfigs(): Promise<ConfigCheckResult> {
    const workspacePath = this.getWorkspacePath();
    const checks = [
      {
        name: 'publish-config.json',
        path: path.join(workspacePath, '.releasePlan', 'publish-config.json'),
        required: true
      },
      {
        name: 'release-notes.json',
        path: path.join(workspacePath, '.releasePlan', 'release-notes.json'),
        required: true
      },
      {
        name: 'smtp-config.json',
        path: path.join(workspacePath, '.releasePlan', 'smtp-config.json'),
        required: true
      }
    ];
    
    const results = await Promise.all(
      checks.map(async (check) => {
        const exists = await fs.pathExists(check.path);
        return {
          ...check,
          exists,
          valid: exists || !check.required,
          error: exists ? undefined : '文件不存在'
        };
      })
    );
    
    const allValid = results.every(r => r.valid);
    
    if (!allValid) {
      throw new ConfigCheckError(results);
    }
    
    this.stateMachine.transition('PACKAGING');
    
    return { valid: true, checks: results };
  }
  
  /**
   * 处理错误
   */
  private handleError(error: unknown): void {
    this.stateMachine.transition('ERROR');
    
    if (error instanceof PublishError) {
      this.logger.error('发布流程错误', error, { type: error.type });
      this.notifyWebview('error', {
        type: error.type,
        message: error.message,
        recoverable: error.recoverable
      });
    } else {
      this.logger.error('未知错误', error as Error);
      this.notifyWebview('error', {
        type: 'UNKNOWN',
        message: (error as Error).message,
        recoverable: false
      });
    }
  }
  
  /**
   * 取消发布流程
   */
  async cancelPublish(): Promise<void> {
    if (!this.stateMachine.canCancel()) {
      return;
    }
    
    this.logger.info('取消发布流程');
    this.stateMachine.reset();
  }
  
  /**
   * 重置发布流程
   */
  reset(): void {
    this.logger.info('重置发布流程');
    this.stateMachine.reset();
  }
  
  /**
   * 获取当前状态
   */
  getState(): PublishState {
    return this.stateMachine.getState();
  }
  
  /**
   * 获取进度
   */
  getProgress(): number {
    return this.stateMachine.getProgress();
  }
  
  /**
   * 获取所有步骤
   */
  getSteps() {
    return this.stateMachine.getAllSteps();
  }
  
  /**
   * 设置 Webview 面板
   */
  setWebviewPanel(panel: vscode.WebviewPanel): void {
    this.webviewPanel = panel;
  }
  
  /**
   * 通知 Webview
   */
  private notifyWebview(type: string, data?: unknown): void {
    if (this.webviewPanel) {
      this.webviewPanel.webview.postMessage({ type, data });
    }
  }
  
  /**
   * 生成流程ID
   */
  private generateFlowId(): string {
    const timestamp = Date.now();
    const random = Math.random().toString(36).substr(2, 9);
    return `flow_${timestamp}_${random}`;
  }
  
  /**
   * 获取工作区路径
   */
  private getWorkspacePath(): string {
    if (vscode.workspace.workspaceFolders && vscode.workspace.workspaceFolders.length > 0) {
      return vscode.workspace.workspaceFolders[0].uri.fsPath;
    }
    throw new Error('没有打开的工作区');
  }
}
```
