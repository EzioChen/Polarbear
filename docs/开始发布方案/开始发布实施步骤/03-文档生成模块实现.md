## 一、目标与范围

### 1.1 目标
实现文档生成模块，能够基于 `release-notes.json` 中的发布说明内容，生成规范化的 Markdown 原文件和 PDF 格式文档，确保文档排版美观、支持文本搜索。

### 1.2 范围
- 读取和解析 `release-notes.json` 配置

- PDF 文档生成（使用 Puppeteer）
- 文档样式模板设计
- 文档缓存管理
- 异常处理机制

### 2.2 实现文档生成模块

**文件路径**：`src/publishFlow/modules/DocGenModule.ts`

**操作内容**：

```typescript
import * as vscode from 'vscode';
import * as path from 'path';
import * as fs from 'fs-extra';
import { marked } from 'marked';
import puppeteer from 'puppeteer';
import { CacheStore } from '../CacheStore';
import { PublishLogger, logger } from '../logger';
import { PublishError, ErrorType } from '../errors';
import { DocumentationOutput } from '../types';

/**
 * SDK版本信息
 */
interface SdkVersion {
  id: string;
  chip: string;
  version: string;
  remark: string;
}

/**
 * 发布说明配置
 */
interface ReleaseNotesConfig {
  version: string;
  createdAt: string;
  updatedAt: string;
  versionLevel: 'A' | 'B' | 'C';
  releaseNotes: string;
  chipRange: string[];
  sdkVersions: SdkVersion[];
  sdkVersion: string;
  appVersion: string;
  remarks: string;
  metadata: {
    createdBy: string;
    author: string;
  };
}

/**
 * 文档生成结果
 */
interface DocGenResult {
  mdPath: string;
  pdfPath: string;
  version: string;
}

/**
 * PDF生成选项
 */
interface PDFOptions {
  format?: 'A4' | 'Letter' | 'Legal';
  margin?: {
    top?: string;
    right?: string;
    bottom?: string;
    left?: string;
  };
  printBackground?: boolean;
}

/**
 * 文档生成模块
 */
export class DocGenModule {
  private configPath: string;
  private workspacePath: string;
  
  constructor(
    private cacheStore: CacheStore,
    private logger: PublishLogger
  ) {
    this.workspacePath = this.getWorkspacePath();
    this.configPath = path.join(
      this.workspacePath,
      '.releasePlan',
      'release-notes.json'
    );
  }
  
  /**
   * 执行文档生成流程
   */
  async execute(flowId: string): Promise<DocumentationOutput> {
    this.logger.info('开始文档生成流程', { flowId });
    
    try {
      // 1. 读取配置
      const config = await this.loadConfig();
      this.logger.info('发布说明配置加载成功', { 
        version: config.appVersion,
        author: config.metadata.author 
      });
      
      // 2. 验证配置
      this.validateConfig(config);
      
      // 3. 生成Markdown
      const mdContent = this.generateMarkdown(config);
      const mdPath = await this.saveMarkdown(mdContent, flowId);
      this.logger.info('Markdown文档生成完成', { mdPath });
      
      // 4. 生成PDF
      const pdfPath = await this.generatePDF(mdContent, flowId, config.appVersion);
      this.logger.info('PDF文档生成完成', { pdfPath });
      
      this.logger.info('文档生成流程完成', { flowId });
      
      return {
        mdPath,
        pdfPath,
        version: config.appVersion
      };
      
    } catch (error) {
      this.logger.error('文档生成失败', error as Error, { flowId });
      throw error;
    }
  }
  
  /**
   * 加载配置文件
   */
  private async loadConfig(): Promise<ReleaseNotesConfig> {
    if (!(await fs.pathExists(this.configPath))) {
      throw new PublishError(
        ErrorType.CONFIG_NOT_FOUND,
        `发布说明配置文件不存在: ${this.configPath}`,
        { configPath: this.configPath },
        true
      );
    }
    
    try {
      const content = await fs.readFile(this.configPath, 'utf-8');
      const config: ReleaseNotesConfig = JSON.parse(content);
      return config;
    } catch (error) {
      throw new PublishError(
        ErrorType.CONFIG_INVALID,
        `发布说明配置解析失败: ${(error as Error).message}`,
        { configPath: this.configPath },
        true
      );
    }
  }
  
  /**
   * 验证配置
   */
  private validateConfig(config: ReleaseNotesConfig): void {
    const requiredFields = ['appVersion', 'releaseNotes', 'sdkVersion'];
    const missingFields = requiredFields.filter(field => !config[field as keyof ReleaseNotesConfig]);
    
    if (missingFields.length > 0) {
      throw new PublishError(
        ErrorType.CONFIG_INVALID,
        `配置缺少必填字段: ${missingFields.join(', ')}`,
        { missingFields },
        true
      );
    }
  }
  
  /**
   * 生成Markdown内容
   */
  private generateMarkdown(config: ReleaseNotesConfig): string {
    const versionLevelText = {
      'A': '重大版本 (Major)',
      'B': '次要版本 (Minor)',
      'C': '补丁版本 (Patch)'
    };
    
    const chipRangeText = config.chipRange.join(', ');
    
    const sdkVersionsTable = config.sdkVersions.map(v => 
      `| ${v.chip} | ${v.version} | ${v.remark || '-'} |`
    ).join('\n');
    
    return `# 发布说明 - ${config.appVersion}

## 基本信息

| 项目 | 内容 |
|------|------|
| 应用版本 | ${config.appVersion} |
| SDK版本 | ${config.sdkVersion} |
| 版本级别 | ${versionLevelText[config.versionLevel]} |
| 发布日期 | ${new Date(config.createdAt).toLocaleDateString('zh-CN')} |
| 作者 | ${config.metadata.author || '未知'} |

## 适用芯片

${chipRangeText}

## SDK版本详情

| 芯片 | 版本 | 备注 |
|------|------|------|
${sdkVersionsTable}

## 发布说明

${config.releaseNotes}

## 备注

${config.remarks || '无'}

---

*文档生成时间: ${new Date().toLocaleString('zh-CN')}*
*生成工具: PolarBear Release System*
`;
  }
  
  /**
   * 保存Markdown文件
   */
  private async saveMarkdown(content: string, flowId: string): Promise<string> {
    const filename = `release-notes.md`;
    const filePath = this.cacheStore.getPath(filename, flowId);
    
    await fs.writeFile(filePath, content, 'utf-8');
    
    return filePath;
  }
  
  /**
   * 生成PDF文件
   */
  private async generatePDF(
    markdownContent: string,
    flowId: string,
    version: string,
    options: PDFOptions = {}
  ): Promise<string> {
    const defaultOptions: PDFOptions = {
      format: 'A4',
      margin: {
        top: '2cm',
        right: '2cm',
        bottom: '2cm',
        left: '2cm'
      },
      printBackground: true
    };
    
    const pdfOptions = { ...defaultOptions, ...options };
    
    // 转换Markdown为HTML
    const htmlContent = this.generateHTML(markdownContent, version);
    
    // 启动浏览器
    const browser = await puppeteer.launch({
      headless: true,
      args: ['--no-sandbox', '--disable-setuid-sandbox']
    });
    
    try {
      const page = await browser.newPage();
      
      // 设置内容
      await page.setContent(htmlContent, {
        waitUntil: 'networkidle0'
      });
      
      // 生成PDF
      const pdfPath = this.cacheStore.getPath('release-notes.pdf', flowId);
      await page.pdf({
        path: pdfPath,
        format: pdfOptions.format,
        margin: pdfOptions.margin,
        printBackground: pdfOptions.printBackground,
        displayHeaderFooter: true,
        headerTemplate: `
          <div style="font-size: 9px; width: 100%; text-align: center; color: #666;">
            发布说明 - ${version}
          </div>
        `,
        footerTemplate: `
          <div style="font-size: 9px; width: 100%; text-align: center; color: #666;">
            第 <span class="pageNumber"></span> 页 / 共 <span class="totalPages"></span> 页
          </div>
        `
      });
      
      return pdfPath;
    } finally {
      await browser.close();
    }
  }
  
  /**
   * 生成HTML内容
   */
  private generateHTML(markdownContent: string, version: string): string {
    const htmlBody = marked.parse(markdownContent);
    
    return `<!DOCTYPE html>
<html lang="zh-CN">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>发布说明 - ${version}</title>
  <style>
    @page {
      size: A4;
      margin: 2cm;
    }
    
    * {
      box-sizing: border-box;
    }
    
    body {
      font-family: "Microsoft YaHei", "SimHei", "Helvetica Neue", Arial, sans-serif;
      font-size: 12pt;
      line-height: 1.6;
      color: #333;
      max-width: 100%;
      margin: 0;
      padding: 0;
    }
    
    h1 {
      font-size: 24pt;
      color: #1a1a1a;
      border-bottom: 2px solid #0066cc;
      padding-bottom: 10px;
      margin-top: 0;
      page-break-after: avoid;
    }
    
    h2 {
      font-size: 18pt;
      color: #2a2a2a;
      margin-top: 24pt;
      margin-bottom: 12pt;
      page-break-after: avoid;
    }
    
    h3 {
      font-size: 14pt;
      color: #3a3a3a;
      margin-top: 18pt;
      page-break-after: avoid;
    }
    
    p {
      margin: 8pt 0;
      text-align: justify;
    }
    
    table {
      width: 100%;
      border-collapse: collapse;
      margin: 12pt 0;
      page-break-inside: avoid;
    }
    
    th, td {
      border: 1px solid #ddd;
      padding: 8pt;
      text-align: left;
    }
    
    th {
      background-color: #f5f5f5;
      font-weight: bold;
    }
    
    tr:nth-child(even) {
      background-color: #fafafa;
    }
    
    code {
      background-color: #f4f4f4;
      padding: 2px 6px;
      border-radius: 3px;
      font-family: "Consolas", "Monaco", monospace;
      font-size: 10pt;
    }
    
    pre {
      background-color: #f4f4f4;
      padding: 12pt;
      border-radius: 4px;
      overflow-x: auto;
      page-break-inside: avoid;
    }
    
    pre code {
      background: none;
      padding: 0;
    }
    
    ul, ol {
      margin: 8pt 0;
      padding-left: 24pt;
    }
    
    li {
      margin: 4pt 0;
    }
    
    blockquote {
      border-left: 4px solid #0066cc;
      margin: 12pt 0;
      padding: 8pt 16pt;
      background-color: #f9f9f9;
      page-break-inside: avoid;
    }
    
    hr {
      border: none;
      border-top: 1px solid #ddd;
      margin: 16pt 0;
    }
    
    em {
      color: #666;
      font-size: 10pt;
    }
    
    /* 打印优化 */
    @media print {
      body {
        print-color-adjust: exact;
        -webkit-print-color-adjust: exact;
      }
    }
  </style>
</head>
<body>
  ${htmlBody}
</body>
</html>`;
  }
  
  /**
   * 获取工作区路径
   */
  private getWorkspacePath(): string {
    if (vscode.workspace.workspaceFolders && vscode.workspace.workspaceFolders.length > 0) {
      return vscode.workspace.workspaceFolders[0].uri.fsPath;
    }
    throw new Error('没有打开的工作区');
  }
  
  /**
   * 获取生成的文档信息
   */
  async getDocs(flowId: string): Promise<{
    mdPath: string | null;
    pdfPath: string | null;
    mdSize: string;
    pdfSize: string;
  }> {
    const mdPath = this.cacheStore.getPath('release-notes.md', flowId);
    const pdfPath = this.cacheStore.getPath('release-notes.pdf', flowId);
    
    const mdExists = await fs.pathExists(mdPath);
    const pdfExists = await fs.pathExists(pdfPath);
    
    let mdSize = '0 B';
    let pdfSize = '0 B';
    
    if (mdExists) {
      const stats = await fs.stat(mdPath);
      mdSize = this.formatFileSize(stats.size);
    }
    
    if (pdfExists) {
      const stats = await fs.stat(pdfPath);
      pdfSize = this.formatFileSize(stats.size);
    }
    
    return {
      mdPath: mdExists ? mdPath : null,
      pdfPath: pdfExists ? pdfPath : null,
      mdSize,
      pdfSize
    };
  }
  
  /**
   * 格式化文件大小
   */
  private formatFileSize(bytes: number): string {
    const units = ['B', 'KB', 'MB', 'GB'];
    let size = bytes;
    let unitIndex = 0;
    
    while (size >= 1024 && unitIndex < units.length - 1) {
      size /= 1024;
      unitIndex++;
    }
    
    return `${size.toFixed(2)} ${units[unitIndex]}`;
  }
}
```

/**
 * 预览Markdown内容
 */
async previewMarkdown(flowId: string): Promise<string | null> {
  const mdPath = this.cacheStore.getPath('release-notes.md', flowId);
  
  if (!(await fs.pathExists(mdPath))) {
    return null;
  }
  
  return fs.readFile(mdPath, 'utf-8');
}

/**
 * 在VS Code中打开文档
 */
async openDocument(flowId: string, type: 'md' | 'pdf' = 'md'): Promise<void> {
  const filename = type === 'md' ? 'release-notes.md' : 'release-notes.pdf';
  const filePath = this.cacheStore.getPath(filename, flowId);
  
  if (!(await fs.pathExists(filePath))) {
    throw new PublishError(
      ErrorType.FILE_NOT_FOUND,
      `文档不存在: ${filename}`,
      {},
      true
    );
  }
  
  const uri = vscode.Uri.file(filePath);
  await vscode.commands.executeCommand('vscode.open', uri);
}
