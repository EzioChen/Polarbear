### 2.2 实现文件打包模块

**文件路径**：`src/publishFlow/modules/FilePackModule.ts`

**操作内容**：

```typescript
import * as vscode from 'vscode';
import * as path from 'path';
import * as fs from 'fs-extra';
import * as archiver from 'archiver';
import { createWriteStream } from 'fs';
import { CacheStore } from '../CacheStore';
import { PublishLogger, logger } from '../logger';
import { PublishError, FileNotFoundError, ErrorType } from '../errors';
import { PackagingOutput } from '../types';

/**
 * 文件节点接口
 */
interface FileNode {
  id: string;
  type: 'file' | 'folder';
  name: string;
  path: string;
  sourcePath?: string;
  size?: number;
  lastModified?: string;
  children?: FileNode[];
}

/**
 * 发布配置接口
 */
interface PublishConfig {
  version: string;
  createdAt: string;
  updatedAt: string;
  name: string;
  description: string;
  files: FileNode[];
  metadata: {
    totalFiles: number;
    totalFolders: number;
    totalSize: number;
  };
}

/**
 * 文件项
 */
interface FileItem {
  sourcePath: string;
  targetPath: string;
  size: number;
}

/**
 * 打包结果
 */
interface PackResult {
  zipPath: string;
  manifestPath: string;
  fileCount: number;
  totalSize: number;
}

/**
 * 验证结果
 */
interface ValidationResult {
  valid: boolean;
  missingFiles: string[];
  permissionErrors: string[];
  warnings: string[];
}

/**
 * 文件打包模块
 */
export class FilePackModule {
  private configPath: string;
  private workspacePath: string;
  
  constructor(
    private cacheStore: CacheStore,
    private logger: PublishLogger
  ) {
    this.workspacePath = this.getWorkspacePath();
    this.configPath = path.join(
      this.workspacePath,
      '.releasePlan',
      'publish-config.json'
    );
  }
  
  /**
   * 执行打包流程
   */
  async execute(flowId: string): Promise<PackagingOutput> {
    this.logger.info('开始文件打包流程', { flowId });
    
    try {
      // 1. 读取配置
      const config = await this.loadConfig();
      this.logger.info('配置加载成功', { 
        name: config.name, 
        fileCount: config.metadata.totalFiles 
      });
      
      // 2. 验证文件
      const validation = await this.validateFiles(config.files);
      if (!validation.valid) {
        throw this.createValidationError(validation);
      }
      
      // 3. 收集文件
      const files = await this.collectFiles(config.files);
      this.logger.info('文件收集完成', { count: files.length });
      
      // 4. 创建ZIP
      const zipPath = this.cacheStore.getPath('package.zip', flowId);
      const zipResult = await this.createZip(files, zipPath);
      this.logger.info('ZIP压缩完成', { 
        path: zipResult.zipPath,
        size: zipResult.totalSize 
      });
      
      // 5. 创建清单
      const manifestPath = await this.createManifest(
        config,
        files,
        zipResult,
        flowId
      );
      
      this.logger.info('文件打包流程完成', { flowId });
      
      return {
        zipPath: zipResult.zipPath,
        manifestPath,
        fileCount: zipResult.fileCount,
        totalSize: zipResult.totalSize
      };
      
    } catch (error) {
      this.logger.error('文件打包失败', error as Error, { flowId });
      throw error;
    }
  }
  
  /**
   * 加载配置文件
   */
  private async loadConfig(): Promise<PublishConfig> {
    if (!(await fs.pathExists(this.configPath))) {
      throw new PublishError(
        ErrorType.CONFIG_NOT_FOUND,
        `配置文件不存在: ${this.configPath}`,
        { configPath: this.configPath },
        true
      );
    }
    
    try {
      const content = await fs.readFile(this.configPath, 'utf-8');
      const config: PublishConfig = JSON.parse(content);
      
      // 验证配置结构
      this.validateConfigStructure(config);
      
      return config;
    } catch (error) {
      if (error instanceof PublishError) {
        throw error;
      }
      throw new PublishError(
        ErrorType.CONFIG_INVALID,
        `配置文件解析失败: ${(error as Error).message}`,
        { configPath: this.configPath },
        true
      );
    }
  }
  
  /**
   * 验证配置结构
   */
  private validateConfigStructure(config: unknown): void {
    const c = config as PublishConfig;
    
    if (!c.version) {
      throw new PublishError(
        ErrorType.CONFIG_INVALID,
        '配置缺少 version 字段',
        {},
        true
      );
    }
    
    if (!Array.isArray(c.files)) {
      throw new PublishError(
        ErrorType.CONFIG_INVALID,
        '配置缺少 files 数组或格式不正确',
        {},
        true
      );
    }
  }
  
  /**
   * 验证文件
   */
  private async validateFiles(nodes: FileNode[]): Promise<ValidationResult> {
    const result: ValidationResult = {
      valid: true,
      missingFiles: [],
      permissionErrors: [],
      warnings: []
    };
    
    const validateNode = async (node: FileNode): Promise<void> => {
      if (node.type === 'file' && node.sourcePath) {
        // 检查文件是否存在
        const exists = await fs.pathExists(node.sourcePath);
        if (!exists) {
          result.missingFiles.push(node.sourcePath);
          result.valid = false;
          return;
        }
        
        // 检查文件可读性
        try {
          await fs.access(node.sourcePath, fs.constants.R_OK);
        } catch {
          result.permissionErrors.push(node.sourcePath);
          result.valid = false;
        }
        
        // 检查文件大小
        const stats = await fs.stat(node.sourcePath);
        const maxSize = 100 * 1024 * 1024; // 100MB
        if (stats.size > maxSize) {
          result.warnings.push(
            `文件过大: ${node.name} (${this.formatFileSize(stats.size)})`
          );
        }
      } else if (node.type === 'folder' && node.children) {
        // 递归验证子节点
        for (const child of node.children) {
          await validateNode(child);
        }
      }
    };
    
    for (const node of nodes) {
      await validateNode(node);
    }
    
    return result;
  }
  
  /**
   * 创建验证错误
   */
  private createValidationError(result: ValidationResult): PublishError {
    const messages: string[] = [];
    
    if (result.missingFiles.length > 0) {
      messages.push(`缺失文件: ${result.missingFiles.length}个`);
    }
    
    if (result.permissionErrors.length > 0) {
      messages.push(`权限错误: ${result.permissionErrors.length}个`);
    }
    
    return new PublishError(
      ErrorType.FILE_NOT_FOUND,
      `文件验证失败: ${messages.join(', ')}`,
      {
        missingFiles: result.missingFiles,
        permissionErrors: result.permissionErrors,
        warnings: result.warnings
      },
      true
    );
  }
  
  /**
   * 收集文件
   */
  private async collectFiles(nodes: FileNode[]): Promise<FileItem[]> {
    const files: FileItem[] = [];
    
    const collectNode = async (node: FileNode): Promise<void> => {
      if (node.type === 'file' && node.sourcePath) {
        const stats = await fs.stat(node.sourcePath);
        files.push({
          sourcePath: node.sourcePath,
          targetPath: node.path,
          size: stats.size
        });
      } else if (node.type === 'folder' && node.children) {
        for (const child of node.children) {
          await collectNode(child);
        }
      }
    };
    
    for (const node of nodes) {
      await collectNode(node);
    }
    
    return files;
  }
  
  /**
   * 创建ZIP压缩包
   */
  private async createZip(
    files: FileItem[],
    outputPath: string
  ): Promise<PackResult> {
    return new Promise((resolve, reject) => {
      const output = createWriteStream(outputPath);
      const archive = archiver('zip', {
        zlib: { level: 9 } // 最高压缩级别
      });
      
      let totalSize = 0;
      
      output.on('close', () => {
        resolve({
          zipPath: outputPath,
          manifestPath: '', // 稍后设置
          fileCount: files.length,
          totalSize: archive.pointer()
        });
      });
      
      archive.on('error', (err) => {
        reject(new PublishError(
          ErrorType.STEP_EXECUTION_FAILED,
          `ZIP压缩失败: ${err.message}`,
          {},
          true
        ));
      });
      
      archive.on('warning', (err) => {
        this.logger.warn('ZIP压缩警告', { message: err.message });
      });
      
      archive.pipe(output);
      
      // 添加文件到压缩包
      for (const file of files) {
        archive.file(file.sourcePath, { name: file.targetPath });
        totalSize += file.size;
      }
      
      archive.finalize();
    });
  }
  
  /**
   * 创建打包清单
   */
  private async createManifest(
    config: PublishConfig,
    files: FileItem[],
    zipResult: PackResult,
    flowId: string
  ): Promise<string> {
    const manifest = {
      version: config.version,
      createdAt: new Date().toISOString(),
      sourceConfig: this.configPath,
      flowId,
      files: files.map(f => ({
        sourcePath: f.sourcePath,
        targetPath: f.targetPath,
        size: f.size
      })),
      totalSize: zipResult.totalSize,
      fileCount: zipResult.fileCount,
      zipPath: zipResult.zipPath
    };
    
    const manifestPath = this.cacheStore.getPath('manifest.json', flowId);
    await fs.writeFile(manifestPath, JSON.stringify(manifest, null, 2));
    
    this.logger.info('打包清单已创建', { manifestPath });
    
    return manifestPath;
  }
  
  /**
   * 格式化文件大小
   */
  private formatFileSize(bytes: number): string {
    const units = ['B', 'KB', 'MB', 'GB'];
    let size = bytes;
    let unitIndex = 0;
    
    while (size >= 1024 && unitIndex < units.length - 1) {
      size /= 1024;
      unitIndex++;
    }
    
    return `${size.toFixed(2)} ${units[unitIndex]}`;
  }
  
  /**
   * 获取工作区路径
   */
  private getWorkspacePath(): string {
    if (vscode.workspace.workspaceFolders && vscode.workspace.workspaceFolders.length > 0) {
      return vscode.workspace.workspaceFolders[0].uri.fsPath;
    }
    throw new Error('没有打开的工作区');
  }
  
  /**
   * 获取打包统计信息
   */
  async getStats(flowId: string): Promise<{
    fileCount: number;
    totalSize: string;
    zipSize: string;
  } | null> {
    const manifestPath = this.cacheStore.getPath('manifest.json', flowId);
    
    if (!(await fs.pathExists(manifestPath))) {
      return null;
    }
    
    try {
      const content = await fs.readFile(manifestPath, 'utf-8');
      const manifest = JSON.parse(content);
      
      const zipPath = manifest.zipPath;
      let zipSize = 0;
      if (await fs.pathExists(zipPath)) {
        const stats = await fs.stat(zipPath);
        zipSize = stats.size;
      }
      
      return {
        fileCount: manifest.fileCount,
        totalSize: this.formatFileSize(manifest.totalSize),
        zipSize: this.formatFileSize(zipSize)
      };
    } catch {
      return null;
    }
  }
}
```
