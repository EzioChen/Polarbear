<script setup lang="ts">
import { ref, reactive, watch, onMounted } from 'vue';
import type { ReleaseNotesConfig, VersionLevelInfo } from '../types/releaseNotes';
import MarkdownEditor from '../components/releaseNotes/MarkdownEditor.vue';
import VersionLevelSelector from '../components/releaseNotes/VersionLevelSelector.vue';
import ChipRangeInput from '../components/releaseNotes/ChipRangeInput.vue';
import SdkVersionTable from '../components/releaseNotes/SdkVersionTable.vue';
import VersionInput from '../components/releaseNotes/VersionInput.vue';

// 版本等级选项
const versionLevelOptions: VersionLevelInfo[] = [
  {
    level: 'A',
    name: '问题修复',
    description: '修复严重问题，客户必须修复',
    required: true
  },
  {
    level: 'B',
    name: '性能优化',
    description: '性能改进，客户可选择修复',
    required: false
  },
  {
    level: 'C',
    name: '功能增加',
    description: '新增功能，客户可选择修复',
    required: false
  }
];

// 表单数据
const form = reactive<ReleaseNotesConfig>({
  version: '1.0.0',
  createdAt: new Date().toISOString(),
  updatedAt: new Date().toISOString(),
  versionLevel: 'C',
  releaseNotes: '',
  chipRange: [],
  sdkVersions: [],
  sdkVersion: '',
  appVersion: '',
  remarks: '',
  metadata: {
    createdBy: '',
    author: ''
  }
});

// 保存状态
const saveStatus = ref<{ type: 'saving' | 'saved' | 'error'; message: string } | null>(null);

// 防抖定时器
let autoSaveTimer: NodeJS.Timeout | null = null;

// 初始化
onMounted(() => {
  // 获取当前配置
  vscode.postMessage({
    type: 'getReleaseNotesConfig'
  });
});

// 监听表单变化，触发自动保存
watch(
  () => ({ ...form }),
  () => {
    saveStatus.value = { type: 'saving', message: '未保存' };
    debouncedAutoSave();
  },
  { deep: true }
);

// 防抖自动保存（500ms）
const debouncedAutoSave = () => {
  if (autoSaveTimer) clearTimeout(autoSaveTimer);
  autoSaveTimer = setTimeout(() => {
    handleAutoSave();
  }, 500);
};

// 自动保存
const handleAutoSave = () => {
  saveStatus.value = { type: 'saving', message: '保存中...' };
  
  vscode.postMessage({
    type: 'autoSaveReleaseNotes',
    payload: { ...form }
  });
};

// 手动保存
const handleManualSave = () => {
  saveStatus.value = { type: 'saving', message: '保存中...' };
  
  vscode.postMessage({
    type: 'saveReleaseNotes',
    payload: { ...form }
  });
};

// 监听来自 extension 的消息
window.addEventListener('message', (event) => {
  const message = event.data;
  
  switch (message.type) {
    case 'releaseNotesConfigUpdated':
      // 更新表单数据
      Object.assign(form, message.payload);
      break;
      
    case 'releaseNotesSaved':
      saveStatus.value = { type: 'saved', message: '已保存' };
      // 3秒后清除状态
      setTimeout(() => {
        saveStatus.value = null;
      }, 3000);
      break;
      
    case 'releaseNotesSaveError':
      saveStatus.value = { type: 'error', message: '保存失败' };
      break;
  }
});

// 格式化日期
const formatDate = (dateStr: string) => {
  const date = new Date(dateStr);
  return date.toLocaleString('zh-CN', {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
    hour: '2-digit',
    minute: '2-digit'
  });
};
</script>

## 4. 更新 Webview HTML 模板

### 4.1 修改 `src/webview/templates/releaseNotesTemplate.ts`

```typescript
import * as vscode from 'vscode';
import * as path from 'path';

export function getReleaseNotesHtml(
  webview: vscode.Webview,
  extensionUri: vscode.Uri
): string {
  // 获取资源路径
  const scriptUri = webview.asWebviewUri(
    vscode.Uri.joinPath(extensionUri, 'dist', 'webview', 'assets', 'index.js')
  );
  const styleUri = webview.asWebviewUri(
    vscode.Uri.joinPath(extensionUri, 'dist', 'webview', 'assets', 'index.css')
  );

  const nonce = getNonce();

  return `<!DOCTYPE html>
<html lang="zh-CN">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="Content-Security-Policy" content="default-src 'none'; style-src ${webview.cspSource} 'unsafe-inline'; script-src 'nonce-${nonce}'; img-src ${webview.cspSource} data: https:;">
  <title>发布说明</title>
  <link rel="stylesheet" href="${styleUri}">
</head>
<body>
  <div id="app"></div>
  <script nonce="${nonce}">
    // 声明 vscode API
    const vscode = acquireVsCodeApi();
    
    // 设置初始路由
    window.__INITIAL_ROUTE__ = '/release-notes';
  </script>
  <script type="module" nonce="${nonce}" src="${scriptUri}"></script>
</body>
</html>`;
}

function getNonce(): string {
  let text = '';
  const possible = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
  for (let i = 0; i < 32; i++) {
    text += possible.charAt(Math.floor(Math.random() * possible.length));
  }
  return text;
}
```

## 常见问题排查
