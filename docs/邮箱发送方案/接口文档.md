## 二、EmailService 接口

### 2.1 initialize

**功能**: 初始化邮件服务，加载并验证配置

**签名**:
```typescript
async initialize(configPath?: string): Promise<void>
```

**参数**:

| 参数名 | 类型 | 必填 | 说明 |
|-------|------|------|------|
| configPath | string | 否 | 配置文件路径，默认使用 `.releasePlan/smtp-config.json` |

**返回值**: `Promise<void>`

**异常**:

| 异常类型 | 说明 |
|---------|------|
| CONFIG_NOT_FOUND | 配置文件不存在 |
| CONFIG_INVALID | 配置文件格式错误 |
| CONFIG_MISSING_FIELD | 缺少必填配置项 |

**示例**:
```typescript
const emailService = EmailService.getInstance();

try {
  await emailService.initialize('./config/smtp-config.json');
  console.log('邮件服务初始化成功');
} catch (error) {
  console.error('初始化失败:', error.message);
}
```

### 2.2 sendEmail

**功能**: 发送邮件

**签名**:
```typescript
async sendEmail(options: SendEmailOptions): Promise<SendResult>
```

**参数**:

| 参数名 | 类型 | 必填 | 说明 |
|-------|------|------|------|
| options | SendEmailOptions | 是 | 邮件发送选项 |

**SendEmailOptions 详情**:

```typescript
interface SendEmailOptions {
  // 收件人，支持单个或多个
  to: string | string[];
  
  // 抄送，可选
  cc?: string | string[];
  
  // 密送，可选
  bcc?: string | string[];
  
  // 邮件主题
  subject: string;
  
  // 纯文本内容，与 html 至少选填一个
  text?: string;
  
  // HTML 内容，与 text 至少选填一个
  html?: string;
  
  // 附件列表，可选
  attachments?: Attachment[];
  
  // 优先级，默认 'normal'
  priority?: 'high' | 'normal' | 'low';
  
  // 自定义邮件头，可选
  headers?: Record<string, string>;
}
```

**Attachment 详情**:

```typescript
interface Attachment {
  // 文件名（必填）
  filename: string;
  
  // 文件内容，以下方式三选一
  content?: Buffer | string | Readable;
  path?: string;
  href?: string;
  
  // 内容类型，如 'application/pdf'
  contentType?: string;
  
  // 内容编码，如 'base64'
  encoding?: string;
  
  // 内容处理方式
  contentDisposition?: 'attachment' | 'inline';
  
  // CID，用于内嵌图片
  cid?: string;
}
```

**返回值**: `Promise<SendResult>`

**SendResult 详情**:

```typescript
interface SendResult {
  // 发送状态
  success: boolean;
  
  // 消息ID（成功时返回）
  messageId?: string;
  
  // 发送时间戳（ISO 8601格式）
  timestamp: string;
  
  // SMTP服务器响应信息
  response?: string;
  
  // 错误信息（失败时返回）
  error?: EmailError;
  
  // 成功接收的收件人列表
  accepted: string[];
  
  // 被拒绝的收件人列表
  rejected: string[];
  
  // 发送耗时（毫秒）
  duration: number;
}
```

**示例**:
```typescript
// 发送纯文本邮件
const result = await emailService.sendEmail({
  to: 'user@example.com',
  subject: '测试邮件',
  text: '这是一封测试邮件'
});

// 发送HTML邮件
const result = await emailService.sendEmail({
  to: ['user1@example.com', 'user2@example.com'],
  cc: 'manager@example.com',
  subject: '发布通知',
  html: '<h1>版本发布</h1><p>新版本已发布</p>',
  priority: 'high'
});

// 发送带附件的邮件
const result = await emailService.sendEmail({
  to: 'user@example.com',
  subject: '报告',
  text: '请查收附件',
  attachments: [
    { filename: 'report.pdf', path: '/path/to/report.pdf' },
    { filename: 'data.xlsx', content: buffer }
  ]
});

// 处理发送结果
if (result.success) {
  console.log(`发送成功，消息ID: ${result.messageId}`);
  console.log(`耗时: ${result.duration}ms`);
} else {
  console.error(`发送失败: ${result.error?.message}`);
}
```

### 2.3 validateConfig

**功能**: 验证当前配置是否有效

**签名**:
```typescript
async validateConfig(): Promise<ValidationResult>
```

**参数**: 无

**返回值**: `Promise<ValidationResult>`

**ValidationResult 详情**:

```typescript
interface ValidationResult {
  // 验证是否通过
  valid: boolean;
  
  // 错误列表
  errors: ValidationError[];
  
  // 警告列表
  warnings: ValidationWarning[];
}

interface ValidationError {
  // 错误字段路径
  field: string;
  
  // 错误描述
  message: string;
  
  // 错误代码
  code: string;
}

interface ValidationWarning {
  // 警告字段路径
  field: string;
  
  // 警告描述
  message: string;
  
  // 建议
  suggestion?: string;
}
```

**示例**:
```typescript
const result = await emailService.validateConfig();

if (result.valid) {
  console.log('配置验证通过');
  if (result.warnings.length > 0) {
    console.warn('警告:', result.warnings);
  }
} else {
  console.error('配置错误:');
  result.errors.forEach(error => {
    console.error(`  ${error.field}: ${error.message}`);
  });
}
```

interface SMTPConfig {

    host: string;
    port: number;

}

### 5.2 错误处理示例

```typescript
import { EmailErrorCode } from './emailService';

async function sendWithErrorHandling() {
  try {
    const result = await emailService.sendEmail({
      to: 'user@example.com',
      subject: '测试',
      text: '内容'
    });
    
    if (!result.success) {
      switch (result.error?.code) {
        case EmailErrorCode.AUTH_INVALID_CREDENTIALS:
          console.error('用户名或密码错误');
          // 提示用户重新输入凭证
          break;
        case EmailErrorCode.CONNECTION_TIMEOUT:
          console.error('连接超时，请检查网络');
          break;
        case EmailErrorCode.RECIPIENT_INVALID:
          console.error('收件人地址无效');
          break;
        default:
          console.error('发送失败:', result.error?.message);
      }
    }
  } catch (error) {
    console.error('异常:', error);
  }
}
```

## 七、使用示例

### 7.1 完整发送流程

```typescript
import { EmailService } from './emailService';

async function main() {
  // 1. 获取服务实例
  const emailService = EmailService.getInstance();
  
  // 2. 初始化服务
  try {
    await emailService.initialize();
  } catch (error) {
    console.error('初始化失败:', error);
    return;
  }
  
  // 3. 验证配置
  const validation = await emailService.validateConfig();
  if (!validation.valid) {
    console.error('配置无效:', validation.errors);
    return;
  }
  
  // 4. 测试连接
  const connection = await emailService.testConnection();
  if (!connection.success) {
    console.error('连接失败:', connection.error);
    return;
  }
  
  // 5. 发送邮件
  const result = await emailService.sendEmail({
    to: 'recipient@example.com',
    subject: '测试邮件',
    html: '<h1>Hello</h1><p>This is a test email.</p>',
    attachments: [
      { filename: 'document.pdf', path: '/path/to/document.pdf' }
    ]
  });
  
  // 6. 处理结果
  if (result.success) {
    console.log('✅ 发送成功');
    console.log(`消息ID: ${result.messageId}`);
    console.log(`耗时: ${result.duration}ms`);
  } else {
    console.error('❌ 发送失败:', result.error?.message);
  }
}

main();
```

### 7.2 与 VS Code 扩展集成

```typescript
import * as vscode from 'vscode';
import { EmailService } from './emailService';

export function activate(context: vscode.ExtensionContext) {
  // 注册发送邮件命令
  const disposable = vscode.commands.registerCommand(
    'polarbear.sendEmail',
    async () => {
      const emailService = EmailService.getInstance();
      
      // 确保已初始化
      try {
        await emailService.initialize();
      } catch (error) {
        vscode.window.showErrorMessage('邮件服务初始化失败');
        return;
      }
      
      // 获取收件人
      const to = await vscode.window.showInputBox({
        prompt: '收件人邮箱',
        placeHolder: 'example@example.com'
      });
      
      if (!to) return;
      
      // 获取主题
      const subject = await vscode.window.showInputBox({
        prompt: '邮件主题',
        placeHolder: '请输入主题'
      });
      
      if (!subject) return;
      
      // 发送邮件
      const result = await emailService.sendEmail({
        to,
        subject,
        text: '从 VS Code 发送的邮件'
      });
      
      if (result.success) {
        vscode.window.showInformationMessage('邮件发送成功！');
      } else {
        vscode.window.showErrorMessage(`发送失败: ${result.error?.message}`);
      }
    }
  );
  
  context.subscriptions.push(disposable);
}
```
